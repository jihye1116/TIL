import {useState, useEffect} from 'react';

function Hello(){
  useEffect(()=>{
    console.log("hi");
  },[]);
  return <h1>Hello</h1>;
}
function App() {
  const [showing, setShowing] = useState(0);
  const onClick = () => setShowing((prev) => !prev);
  return <div>
    {showing ? <Hello/> : null}
    <button onClick={onClick}>{showing ?  "hide": "show"}</button>
  </div>
}

export default App;
